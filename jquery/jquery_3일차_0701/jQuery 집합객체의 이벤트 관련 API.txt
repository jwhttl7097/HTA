// .on()메소드에게 이벤트 이름과 그 이벤트가 발생했을 때 실행할 함수를 전달한다.
// jQuery는 선택된 엘리먼트에서 전달받은 이름의 이벤트가 발생할 때마다 전달받은 함수를 실행한다.
$(selector).on('이벤트이름', function(event) {

});

$(selector).click(function(event) {...})

// .one()메소드에게 이벤트 이름과 그 이벤트가 발생했을 때 실행할 함수를 전달한다.
// jQuery는 선택된 엘리먼트에서 전달받은 이름의 이벤트가 발생하면 전달받은 함수를 딱 한번만 실행한다.
$(selector).one('이벤트이름', function(event) {

}) ;

// .off()메소드에게 이벤트처리를 취소할 이벤트명을 전달한다.
// jQuery는 선택된 엘리먼트에 전달받은 이름의 이벤트에 대한 이벤트처리가 지정되어 있으면 이를 폐기한다.
$(selector).off('이벤트이름')

// .trigger()메소드에게 강제 발생할 이벤트명을 전달한다.
// jQuery는 선택된 엘리먼트에서 전달받은 이름의 이벤트를 강제발생 시킨다.
$(selector).trigger('이벤트이름')
